{{{
run make menuconfig
select target system "Broadcom BCM947xx/953xx [2.4]"

run make kernel_menuconfig
select Memory Technology Devices (MTD) -->
    Ram/Rom/Flash chip drivers -->>>

check all "Support X-chip flash interleave"

}}}
